	notai_dest_primitive_setup_country_effect = yes
	notai_dest_primitive_control_space_effect = yes

		every_deposit = {
			limit = {
				is_blocker = yes

			}

		notai_dest_primitive_add_buildings_effect = yes
		notai_dest_primitive_setup_armies_effect = yes
	}
	random_owned_ship = {
		limit = { is_ship_size = primitive_space_station }
		fleet = { destroy_fleet = THIS }
	}
}

notai_dest_primitive_add_country_resources_effect = {
	add_resource = {
		energy = 1000
		minerals = 750
		alloys = 750
		influence = 300
	}
	if = {
		limit = { 
			NOR = {
				is_lithoid = yes
				is_robot_empire = yes
			}
		}
		add_resource = {
			food = 1000
		}
	}
	if = {
		limit = { 
			NOT = {
				has_ethic = ethic_gestalt_consciousness
			}
		}
		add_resource = {
			consumer_goods = 500
		}
	}
}

notai_dest_primitive_add_districts_effect = {
	remove_all_primitive_buildings = yes
	while = {
		count < 3
		add_district = district_mining
		add_district = district_generator
		add_district = district_city
	}
	if = {
		limit = { 
			is_lithoid = yes
		}
		while = {
			count < 3
			add_district = district_mining
		}
	}
	else_if = {
		limit = { 
			is_machine_empire = yes
		}
		add_district = district_generator
	}
	else = {
		while = {
			count < 3
			add_district = district_farming
		}
	}
}
notai_dest_primitive_add_buildings_effect = {
	if = {
		limit = { 
			has_authority = auth_hive_mind
		}
		add_building = building_hive_capital
		add_building = building_hive_node
		add_building = building_research_lab_1
		add_building = building_foundry_1
		add_building = building_spawning_pool
	}
	else_if = {
		limit = {
			has_authority = auth_machine_intelligence
		 }
		 add_building = building_machine_capital
		 add_building = building_robot_assembly_plant
		 add_building = building_uplink_node
		 add_building = building_research_lab_1
		 add_building = building_foundry_1
	}
	else = {		
		add_building = building_capital
		add_building = building_bureaucratic_1
		add_building = building_research_lab_1
		add_building = building_foundry_1
		add_building = building_factory_1
	}
}
notai_dest_primitive_control_space_effect = {	
	if = {
		limit = {
			home_planet = {
				has_observation_outpost = yes
			}
		}
		home_planet = {
			observation_outpost_owner = {
				country_event = { id = primitive.17 }
			}
			observation_outpost = {
				dismantle = yes
			}
			solar_system.starbase = {
				if = {
					limit = { 
						exists = this
					}
					set_owner = prev.owner
				}
			}
		}
	}
}
