dest_uplift_primitives = {
	capital_scope = {
		every_deposit = {
			limit = { is_blocker = yes }
			clear_blocker = yes
		}
		while = {
			limit = { num_pops < 28 }
			create_pop = {
				species = owner
				ethos = owner
			}
		}
	}
	random_owned_ship = {
		limit = { is_ship_size = primitive_space_station }
		destroy_fleet = fleet
	}
}

dest_uplift_add_country_resources_effect = {
	add_resource = {
		energy = 1250
		minerals = 750
		alloys = 750
		influence = 300
	}
	if = {
		limit = {
			NOR = {
				is_lithoid = yes
				is_robot_empire = yes
			}
		}
		add_resource = { food = 500 }
	} else_if = { limit = { is_lithoid = yes }
		add_resource = { minerals = 500 }
	} else = {
		add_resource = { energy = 250 }
	}
	if = {
		limit = {
			is_gestalt = no
		}
		add_resource = { consumer_goods = 500 }
	}
}

dest_uplift_add_post_ftl_buildings_effect = {
	if = { limit = { is_hive_empire = yes }
		add_building = building_hive_capital
		add_building = building_hive_node
		add_building = building_research_lab_1
		add_building = building_foundry_1
		add_building = building_spawning_pool
	} else_if = { limit = { is_machine_empire = yes }
		add_building = building_machine_capital
		add_building = building_robot_assembly_plant
		add_building = building_uplink_node
		add_building = building_research_lab_1

	} else = {
		add_building = building_capital
		add_building = building_bureaucratic_1
		add_building = building_research_lab_1
		add_building = building_foundry_1
		add_building = building_factory_1
	}
}

dest_uplift_control_system_effect = {
	if = {
		limit = {
			home_planet = { has_observation_outpost = yes }
		}
		home_planet = {
			observation_outpost_owner = {
				country_event = { id = primitive.17 }
			}
			observation_outpost = { dismantle = yes }
			solar_system.starbase = {
				if = { limit = { exists = this }
					set_owner = prev.owner
				}
			}
		}
	}
}
